# 简介

MinIO适配是统一对象存储服务接口（UOSSI）的对MinIO服务的客户端包装实现，需引入MinIO驱动及相关依赖包可使用。

本文将提供Shine常用的三种（epboss、eframe3.0、springboot）三种框架的接入示例。组件同样适用于其他Java框架

  /**
   * 将InputStream写入本地文件
   * @param destination 写入本地目录
   * @param input    输入流
   * @throws IOException
   */
  private static void writeToLocal(File destination, InputStream input)
      throws IOException {
    int index;
    byte[] bytes = new byte[1024];
    FileOutputStream downloadFile = new FileOutputStream(destination);
    while ((index = input.read(bytes)) != -1) {
      downloadFile.write(bytes, 0, index);
      downloadFile.flush();
    }
    downloadFile.close();
    input.close();
  }

  private byte[] getBytesFromFile(FileInfo file) throws BusiException {
    byte[] licenseData = null;
    try{
      InputStream in = file.getInputStream();

      int len = 0;
      byte[] buffer = new byte[1024];
      ByteArrayOutputStream baos = new ByteArrayOutputStream();
      while ((len = in.read(buffer)) != -1) {
        baos.write(buffer, 0, len);
      }
      licenseData = baos.toByteArray();
      in.close();
    }catch(Exception e){
      throw new BusiException(getMessage("epframe.license.exception.readlicense","读取文件出错"));
    }
    return licenseData;
  }

    try {
      com.shine.oss.OSSService ossService = ossClient.getOssServiceImpl();
      RetObject<OSSObject> ret = ossService.getObject(bucketName,key);
      byte[] data = readInputStream(ret.getRetValue().getObjectInputStream());
      DownloadFileInfo downloadFileInfo = new DownloadFileInfo(new ByteArrayInputStream(data),ret.getRetValue().getContentSize());
      downloadFileInfo.setFilename(ret.getRetValue().getName());
      return getSuccessResult(downloadFileInfo);
    }catch (Exception e){
      e.printStackTrace();
      return getFailResult(generalMessageFile("下载文件失败",EXCEPTION_MESSAGE_FILE_NAME));
    }
  }

  /**
   * 从输入流中获取数据
   * @return
   * @throws Exception
   */
  private byte[] readInputStream(InputStream fis) throws IOException {
    ByteArrayOutputStream outStream = new ByteArrayOutputStream();
    byte[] buffer = new byte[1024];
    int len = 0;
    while( (len=fis.read(buffer)) != -1 ){
      outStream.write(buffer, 0, len);
    }
    fis.close();
    return outStream.toByteArray();
  }

  /**
   * 生成异常信息的文件
   * @Description
   * @param message

   * @see

  private DownloadFileInfo generalMessageFile(String message, String message_file_name) {
    byte[] exception_content = message.getBytes();
    InputStream is = new ByteArrayInputStream(exception_content);
    DownloadFileInfo exception_file = new DownloadFileInfo(is, exception_content.length);
    exception_file.setFilename(message_file_name);

# 步骤三、OSSClient实例


/**
 * OSSClient的工厂类
 */
public class OSSClientFactory {

    // 对象存储服务的URL
    private static String endpoint = "http://10.168.0.45:5020";

    // Access key 账户名
    private static String accessKey = "admin";

    // Secret key 账户的密码
    private static String secretKey = "admin123456";

    private static OSSClient minIOClient = null;

    /**
     * 获取MinIO存储的OSSClient
     * @return
     */
    public static OSSClient getMinIOClient(){
        if(minIOClient == null){
            MinioImpl minio = new MinioImpl();
            minio.setEndpoint(endpoint);
            minio.setAccessKey(accessKey);
            minio.setSecretKey(secretKey);
            minio.initial();
            minIOClient = new OSSClient(minio);
        }
        return minIOClient;
    }

}

/**
 * UOSSI 处理Action示例
 * 请求URL：${ContextPath}/jsp/oss/index.do?dispatch=searchObjectsAll&bucketName=abc&prefix=1&offset=2&limit=2
 */
public class UOSSIManagerAction extends Action
{
    //获取OSSClient类型（非组件配置）
    public static final String OSS_TYPE = "minIO";
    //本地文件存储跟路径(非组件配置)
    public static final String LOCAL_DIR = "C:\\Users\\jason\\workspace\\shinespace\\tmp\\localdir\\";

    public UOSSIManagerAction()
    {
    }

    /**
     * 根据配置的OSS_TYPE获取指定的OSSSCLIENT
     * @return
     */
    public OSSClient getOSSClient(){
        switch (OSS_TYPE){
            case "javaFileSystem": return OSSClientFactory.getJavaFileSystemClient();
            case "gridFS": return OSSClientFactory.getGridFSClient();
            case "minIO": return OSSClientFactory.getMinIOClient();
            default: return OSSClientFactory.getJavaFileSystemClient();
        }
    }

    public ActionForward execute(ActionMapping actionMapping, ActionForm actionForm, HttpServletRequest request, HttpServletResponse response)
        throws Exception
    {
        ComInput ci = (ComInput)actionForm;
        String dispatch = (String) ci.get("dispatch");
        RetObject retObject = null;
        switch (dispatch){
            //创建桶
            case "createBucket": retObject = createBucket(ci); break;
            //删除桶
            case "deleteBucket": retObject = deleteBucket(ci); break;
            //检索所有桶（返回前200条桶名称信息）
            case "listALLBuckets": retObject = listALLBuckets(); break;
            //检索桶(分页)
            case "listBucketsForPage": retObject = listBucketsForPage(ci);break;
            //判定桶是否存在
            case "bucketExists": retObject = bucketExists(ci);break;
            //保存对象-文件保存模式
            case "putObjectForFile": retObject = putObjectForFile(ci);break;
            //保存对象-byte[]模式
            case "putObjectForByte": retObject = putObjectForByte(ci);break;
            //保存对象-InputStream模式
            case "putObjectForInputStream": retObject = putObjectForInputStream(ci);break;
            //获取指定对象（文件）的元数据信息
            case "getObjectInfo": retObject = getObjectInfo(ci);break;
            //获取指定对象(使用对象InputStream)
            case "getObjectForStream": getObjectForStream(ci,response);return null;
            //获取指定对象(使用输出流)
            case "getObjectForOutputStream": getObjectForOutputStream(ci,response);return null;
            //获取指定对象(输出文件)
            case "getObjectForFile": retObject = getObjectForFile(ci);return null;
            //删除指定对象
            case "deleteObject": retObject = deleteObject(ci);break;
            //查询对象（桶下所有文件）-上限200条
            case "listObjectsAll": retObject = listObjectsAll(ci);break;
            //查询对象（桶下所有文件）-上限200条
            case "searchObjectsAll": retObject = searchObjectsAll(ci);break;
            //分页查询桶
            case "listObjects": retObject = listObjects(ci);break;
            //分页检索桶
            case "searchObjects": retObject = searchObjects(ci);break;
        }
        System.out.println(new Gson().toJson(retObject));

        return actionMapping.findForward("success");
    }

     * @param input    输入流

    private static void writeToLocal(File destination, InputStream input)
            throws IOException {
        int index;
        byte[] bytes = new byte[1024];
        FileOutputStream downloadFile = new FileOutputStream(destination);
        while ((index = input.read(bytes)) != -1) {
            downloadFile.write(bytes, 0, index);

        String bucketName = (String) form.get("bucketName");
        String key = (String) form.get("key");
        OSSService ossService = this.getOSSClient().getOssServiceImpl();
        RetObject<OSSObject> ret = ossService.getObject(bucketName,key);
        InputStream contentStream = ret.getRetValue().getObjectInputStream();
        byte[] buffer = readInputStream(contentStream);
        if (null != buffer && buffer.length > 0) {
            // 清空response
            response.reset();
            // 设置response的Header
            response.addHeader("Content-Disposition", "attachment;filename=" + new String((ret.getRetValue().getName()).getBytes("GBK"), "ISO8859_1"));
            response.addHeader("Content-Length", "" + buffer.length);
            OutputStream toClient = response.getOutputStream();
            response.setContentType("application/octet-stream");
            toClient.write(buffer);
            toClient.flush();
            toClient.close();
            //注意关闭流的操作
            contentStream.close();
        }
    }

    /**
     * 从输入流中获取数据
     * @return
     * @throws Exception
     */

        ByteArrayOutputStream outStream = new ByteArrayOutputStream();

        int len = 0;

            outStream.write(buffer, 0, len);
        }
        fis.close();
        return outStream.toByteArray();
    }
