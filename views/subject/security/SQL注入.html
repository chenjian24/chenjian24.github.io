# 引言

由于EFRAME3.0平台早期开发时，大量使用SQL拼接参数方式执行数据库，存在SQL注入的安全隐患。本文主要目的是在原来代码的基础上使用最小代价，把SQL拼接参数方式改为预编译SQL方式执行。

# 修改说明

原生JDBC中使用PreparedStatemen来预编译SQL语句，但参数占位符都是使用“?”，并对参数注入有顺序要求，可读性差，容易出错，特别在根据参数条件拼接SQL与注入参数时都要增加if判断。所以使用辅助类JdbcHelper中的createStatement(Connection conn, String sql, Map<String, Object> params)方法重新封装了SQL参数的占位符与注入方式，使用起来更方便。参数说明： conn:JDBC连接对象，在EFRAME3.0系统中统一使用EJBUtil.getConnection()方法获取。 sql:执行的SQL语句，参数占位符用“:xxxx”表示，xxxx为参数名称。 params:SQL语句中参数占位符对应的参数值集合，Map的Key为参数名，Value为参数值。

# 用例说明

# 执行无参数SQL

	Connection conn = null;
	PreparedStatement ps = null;
	ResultSet rs = null;
	String sql = "SELECT ROLE_ID,ROLE_NAME,NOTE FROM DEP_ROLE";
	try {
		conn = EJBUtil.getConnection();
		ps = JdbcHelper.createStatement(conn, sql, null);
		rs = ps.executeQuery();
		comOutput = RSUtil.setOutput(rs);
	} catch(Exception e){
		comOutput = RSUtil.setOutput(-1, e.toString(), null);
		EJBUtil.errorOut(e);
	} finally {
		CommUtil.closeResultSet(rs);
		CommUtil.closeStatement(ps);
		CommUtil.closeConnection(conn);
	}

# IN条件参数

	String roleIds = (String)this.comInput.getPm("ROLE_IDS");
	String roleNames = (String)this.comInput.getPm("ROLE_NAMES");
	Connection conn = null;
	PreparedStatement ps = null;
	ResultSet rs = null;
	String sql = "SELECT ROLE_ID,ROLE_NAME,NOTE FROM DEP_ROLE WHERE 1=1";
	Map<String, Object> params = new HashMap<String, Object>();
	try {
		if (StringUtils.isNotBlank(roleIds)) {
			sql += " AND ROLE_ID in (:ROLE_IDS)";
		    //固定数组值设置方式
	//	    params.put("ROLE_IDS", new Integer[] {11, 12, 13});
			//前端ROLE_IDS字符格式为：11,12,13
		    params.put("ROLE_IDS", JdbcHelper.splitIntValue(roleIds));
		}
		if (StringUtils.isNotBlank(roleNames)) {
			sql += " AND ROLE_NAME in (:ROLE_NAMES)";
		    //固定数组值设置方式
	//	    params.put("ROLE_IDS", new String[] {"ID01", "ID02", "ID03"});
			//前端ROLE_IDS字符格式为：ID01,ID02,ID03
	//		params.put("ROLE_NAMES", roleNames.split(","));
			//前端ROLE_IDS字符格式为：'ID01','ID02','ID03'
		    params.put("ROLE_NAMES", JdbcHelper.splitStrValue(roleNames));
		}
		conn = EJBUtil.getConnection();
		ps = JdbcHelper.createStatement(conn, sql, params);
		rs = ps.executeQuery();
		comOutput = RSUtil.setOutput(rs);
	} catch(Exception e){
		comOutput = RSUtil.setOutput(-1, e.toString(), null);
		EJBUtil.errorOut(e);
	} finally {
		CommUtil.closeResultSet(rs);
		CommUtil.closeStatement(ps);
		CommUtil.closeConnection(conn);
	}

# 注意事项

# 确保JDBC资源都关闭

要确保方法中使用的Connection、Statement、ResultSet等对象最终都关闭，必须在finally中关闭。因为每调用一次JdbcHelper.createStatement方法，都会生成一新的PreparedStatement对象，所以在一个方法中如果执行多条语句多次调用JdbcHelper.createStatement时必须申明多个PreparedStatement及ResultSet变量，并确定最终都关闭，例如：

	String roleName = (String)this.comInput.getPm("ROLE_NAMES");
	Connection conn = null;
	PreparedStatement ps = null;
	PreparedStatement ps1 = null;
	ResultSet rs1 = null;
	String sql = "UPDATE DEP_ROLE SET NOTE=:NOTE FROM DEP_ROLE WHERE ROLE_ID=:ROLE_ID";
	Map<String, Object> params = new HashMap<String, Object>();
	try {
		conn = EJBUtil.getConnection();
		params.put("NOTE", "AAA");
		params.put("ROLE_ID", 1);
		ps = JdbcHelper.createStatement(conn, sql, params);
		ps.executeUpdate();
		
		sql = "SELECT ROLE_ID,ROLE_NAME,NOTE FROM DEP_ROLE WHERE ROLE_NAME=:ROLE_NAME";
		params.clear();
		params.put("ROLE_NAME", roleName);
		//如果复用ps变量，就是导致前一个PreparedStatement未关闭
	//	ps = JdbcHelper.createStatement(conn, sql, params);
		ps1 = JdbcHelper.createStatement(conn, sql, params);
		rs1 = ps1.executeQuery();
		
		comOutput = RSUtil.setOutput(rs1);
	} catch(Exception e){
		comOutput = RSUtil.setOutput(-1, e.toString(), null);
		EJBUtil.errorOut(e);
	} finally {
		CommUtil.closeResultSet(rs1);
		CommUtil.closeStatement(ps);
		CommUtil.closeStatement(ps1);
		CommUtil.closeConnection(conn);
	}

# 禁止拼接参数中的SQL片段

在EFRAME改造过程中，如果发现原来代码中存在前端页面中直接传SQL片段到后台进行拼接的情况，需要重新改造。例如：

//比如前面传过来ROLE_WHERE参数值为： WHERE ROLE_ID=1
String where = (String)this.comInput.getPm("ROLE_WHERE");
String sql = "SELECT ROLE_ID,ROLE_NAME,NOTE FROM DEP_ROLE ";

if (StringUtils.isNotBlank(where)) {
	sql += where;
}
